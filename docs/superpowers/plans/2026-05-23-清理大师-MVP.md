```js
// src/LevelRunner.js
import DirtLayer from './DirtLayer.js'

export default class LevelRunner {
  constructor(levelData, shoeRect, callbacks = {}) {
    this.level = levelData
    this.shoeRect = shoeRect
    this.callbacks = callbacks
    this.stepIndex = 0
    this.selectedTool = null
    this.startedAt = Date.now()
    this.stepStartedAt = Date.now()
    this._dirt = null
    this._objectImg = null
    this._completed = false
    this._loadObject()
  }

  get currentStep() {
    return this.level.steps[this.stepIndex]
  }

  get totalSteps() {
    return this.level.steps.length
  }

  // 全关合成进度 = (已完成步骤数 + 当前步骤擦净比例) / 总步骤数
  get overallProgress() {
    if (this._completed) return 1
    const stepRatio = this._dirt ? this._dirt.getCleanRatio(Date.now()) : 0
    return Math.min(1, (this.stepIndex + Math.min(stepRatio, 1)) / this.totalSteps)
  }

  get currentStepProgress() {
    return this._dirt ? this._dirt.getCleanRatio(Date.now()) : 0
  }

  get usedTools() {
    const set = new Set()
    for (const s of this.level.steps) set.add(s.tool)
    return [...set]
  }

  get dirtCanvas() {
    return this._dirt ? this._dirt.canvas : null
  }

  get objectImage() {
    return this._objectImg
  }

  setSelectedTool(toolId) {
    this.selectedTool = toolId
  }

  erase(x, y) {
    if (this._completed || !this._dirt) return
    const required = this.currentStep.tool
    if (this.selectedTool !== required) {
      if (this.callbacks.onWrongTool) {
        this.callbacks.onWrongTool({ expected: required, got: this.selectedTool })
      }
      return
    }
    this._dirt.erase(x, y)
  }

  tick(now) {
    if (this._completed || !this._dirt) return
    const ratio = this._dirt.getCleanRatio(now)
    if (ratio >= this.currentStep.passThreshold) {
      this._advanceStep()
    }
  }

  destroy() {
    this._dirt = null
    this._objectImg = null
  }

  _loadObject() {
    const img = wx.createImage()
    img.onload = () => {
      this._objectImg = img
      this._createStepLayer(0)
      if (this.callbacks.onReady) this.callbacks.onReady()
    }
    img.src = this.level.objectImage
  }

  _createStepLayer(index) {
    const step = this.level.steps[index]
    const layer = new DirtLayer(this.shoeRect.canvasW, this.shoeRect.canvasH, step.layer, this.shoeRect.rect)
    if (this._objectImg) layer.setMask(this._objectImg)
    this._dirt = layer
  }

  _advanceStep() {
    const now = Date.now()
    const stepDuration = now - this.stepStartedAt
    const finishedIndex = this.stepIndex
    if (this.callbacks.onStepComplete) {
      this.callbacks.onStepComplete({ stepIndex: finishedIndex, durationMs: stepDuration })
    }
    if (this.stepIndex + 1 < this.totalSteps) {
      this.stepIndex++
      this.stepStartedAt = now
      this._createStepLayer(this.stepIndex)
    } else {
      this._completed = true
      const totalDuration = now - this.startedAt
      if (this.callbacks.onLevelComplete) {
        this.callbacks.onLevelComplete({ durationMs: totalDuration })
      }
    }
  }
}
```

注意:`shoeRect` 这里是对象 `{ rect, canvasW, canvasH }`。

```js
// src/GameScene.js
import LevelRunner from './LevelRunner.js'
import UI from './UI.js'
import LEVELS from '../data/levels.js'

export default class GameScene {
  constructor(canvas) {
    this.canvas = canvas
    this.ctx = canvas.getContext('2d')
    this.width = canvas.width
    this.height = canvas.height
    this.levelIndex = 0
    this.runner = null
    this.ui = new UI(this.ctx, this.width, this.height)
    this.uiState = {
      wrongToolHintAt: 0,
      stepAdvancedAt: 0,
      showWin: false,
      isLastLevel: false
    }
    this._enterLevel(this.levelIndex)
    this._bindTouch()
    this._loop()
  }

  _shoeRect() {
    return { x: 0, y: 56, w: this.width, h: this.height - 56 - 100 }
  }

  _computeShoeRect(img) {
    const area = this._shoeRect()
    const imgRatio = img.width / img.height
    const areaRatio = area.w / area.h
    let w, h
    if (imgRatio > areaRatio) { w = area.w; h = area.w / imgRatio }
    else { h = area.h; w = area.h * imgRatio }
    return { x: area.x + (area.w - w) / 2, y: area.y + (area.h - h) / 2, w, h }
  }

  _enterLevel(index) {
    if (this.runner) this.runner.destroy()
    const level = LEVELS[index]
    this.uiState.showWin = false
    this.uiState.isLastLevel = (index === LEVELS.length - 1)
    const initRect = this._shoeRect()
    this.runner = new LevelRunner(level, { rect: initRect, canvasW: this.width, canvasH: this.height }, {
      onReady: () => {
        const newRect = this._computeShoeRect(this.runner.objectImage)
        this.runner.shoeRect = { rect: newRect, canvasW: this.width, canvasH: this.height }
        this.runner._createStepLayer(this.runner.stepIndex)
      },
      onWrongTool: () => { this.uiState.wrongToolHintAt = Date.now() },
      onStepComplete: () => { this.uiState.stepAdvancedAt = Date.now() },
      onLevelComplete: () => { this.uiState.showWin = true }
    })
  }

  _bindTouch() {
    wx.onTouchStart(e => this._onTouchStart(e))
    wx.onTouchMove(e => this._onTouchMove(e))
    wx.onTouchEnd(e => this._onTouchEnd(e))
  }

  _onTouchStart(e) {
    const t = e.changedTouches[0]
    if (this.uiState.showWin) {
      const action = this.ui.hitWinButton(t.clientX, t.clientY)
      if (action === 'replay') this._replayCurrent()
      else if (action === 'next') this._nextLevel()
      else if (action === 'share') { /* 由 Task 12 接入 */ }
      return
    }
    const tools = this.runner ? this.runner.usedTools : []
    const toolHit = this.ui.hitTool(t.clientX, t.clientY, tools)
    if (toolHit !== null) { this.runner.setSelectedTool(toolHit); return }
    this._onTouchMove(e)
  }

  _onTouchMove(e) {
    if (this.uiState.showWin || !this.runner) return
    for (const t of e.touches) this.runner.erase(t.clientX, t.clientY)
  }

  _onTouchEnd() {}

  _replayCurrent() { this._enterLevel(this.levelIndex) }

  _nextLevel() {
    if (this.levelIndex + 1 < LEVELS.length) {
      this.levelIndex++
      this._enterLevel(this.levelIndex)
    } else {
      this._replayCurrent()
    }
  }

  _loop() {
    const step = (ts) => { this._draw(ts); requestAnimationFrame(step) }
    requestAnimationFrame(step)
  }

  _draw(ts) {
    const ctx = this.ctx
    ctx.clearRect(0, 0, this.width, this.height)

    if (this.runner && this.runner.objectImage) {
      const r = this.runner.shoeRect.rect
      ctx.drawImage(this.runner.objectImage, r.x, r.y, r.w, r.h)
    }
    if (this.runner && this.runner.dirtCanvas) {
      ctx.drawImage(this.runner.dirtCanvas, 0, 0)
    }
    if (this.runner) this.runner.tick(ts)

    this.ui.draw({
      level: this.runner ? this.runner.level : null,
      levelIndex: this.levelIndex,
      totalLevels: LEVELS.length,
      overallProgress: this.runner ? this.runner.overallProgress : 0,
      requiredTool: this.runner ? this.runner.currentStep.tool : null,
      selectedTool: this.runner ? this.runner.selectedTool : null,
      usedTools: this.runner ? this.runner.usedTools : [],
      uiState: this.uiState
    })
  }
}
```

  _drawToolbar(state) {
    const ctx = this.ctx
    const usedTools = state.usedTools || []
    const required = state.requiredTool
    const selected = state.selectedTool

    const toolbarY = this.height - BOTTOM_H
    ctx.fillStyle = 'rgba(0,0,0,0.55)'
    ctx.fillRect(0, toolbarY, this.width, BOTTOM_H)

    if (usedTools.length === 0) { this._toolRects = []; return }

    const toolW = 76, toolH = 76, gap = 18
    const totalW = usedTools.length * toolW + (usedTools.length - 1) * gap
    const startX = (this.width - totalW) / 2
    const y = toolbarY + (BOTTOM_H - toolH) / 2

    this._toolRects = usedTools.map((id, i) => ({
      id,
      x: startX + i * (toolW + gap),
      y,
      w: toolW,
      h: toolH
    }))

    for (const r of this._toolRects) {
      const meta = TOOLS_META[r.id]
      const isSelected = selected === r.id
      const isRequired = required === r.id

      ctx.fillStyle = isSelected ? meta.color : 'rgba(255,255,255,0.12)'
      this._roundRect(r.x, r.y, r.w, r.h, 12); ctx.fill()

      if (isRequired && !isSelected) {
        const pulse = 0.6 + 0.3 * Math.sin(Date.now() / 300)
        ctx.strokeStyle = `rgba(255,215,0,${pulse})`
        ctx.lineWidth = 2

## Task 8: 创建 `src/storage.js` 本地进度持久化

**Files:**
- Create: `src/storage.js`

- [ ] **Step 1: 写入 storage**

```js
// src/storage.js
const KEY = 'clean_mini_progress'

const DEFAULT = { unlockedLevel: 1, completedLevels: [], lastPlayedAt: 0 }

export function loadProgress() {
  try {
    const raw = wx.getStorageSync(KEY)
    if (!raw) return { ...DEFAULT }
    const p = typeof raw === 'string' ? JSON.parse(raw) : raw
    return {
      unlockedLevel: typeof p.unlockedLevel === 'number' ? p.unlockedLevel : 1,
      completedLevels: Array.isArray(p.completedLevels) ? p.completedLevels : [],
      lastPlayedAt: typeof p.lastPlayedAt === 'number' ? p.lastPlayedAt : 0
    }
  } catch (e) {
    return { ...DEFAULT }
  }
}

export function saveProgress(progress) {
  try {
    wx.setStorageSync(KEY, JSON.stringify(progress))
  } catch (e) {
    // 静默失败
  }
}

export function markLevelComplete(levelId) {
  const p = loadProgress()
  const completed = p.completedLevels.includes(levelId)
    ? p.completedLevels
    : [...p.completedLevels, levelId]
  const unlockedLevel = Math.max(p.unlockedLevel, levelId + 1)
  const next = { ...p, completedLevels: completed, unlockedLevel, lastPlayedAt: Date.now() }
  saveProgress(next)
  return next
}

export function reset() {
  try { wx.removeStorageSync(KEY) } catch (e) {}
}
```

注意:**immutability** — 所有更新通过 `[...arr, x]` 和 `{ ...obj, k: v }`,不修改原对象。
